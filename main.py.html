# class for my custom build frames.
class DashboardCustomTKframe(customtkinter.CTkFrame):
    def __init__(self, master, width=350, height=150, corner_radius=10, fg_color="white",

        super().__init__(master)
        self.configure(width=width, height=height, corner_radius=corner_radius,

class DashboardCustomTKbutton(customtkinter.CTkButton):

                         image=image)

# checking a password
def passwordcheck(password):
    if len(password) < 8:
        error = 1
        return error
        # raise sqlite3.IntegrityError('Password must be over 8 characters long.')
    elif not re.search(r"[a-z]", password):
        error = 1
        return error
        # raise sqlite3.IntegrityError('Password must have at least 1 lowercase character.')
    elif not re.search(r"[A-Z]", password):
        error = 1
        return error
        # raise sqlite3.IntegrityError('Password must have at least 1 uppercase character.')
    elif not re.search(r"\d", password):
        error = 1
        return error
        # raise sqlite3.IntegrityError('Password must have at least 1 number.')
    elif not re.search(r"[^a-zA-Z0-9]", password):
        error = 1
        return error
        # raise sqlite3.IntegrityError('Password must have at least 1 special character.')
    else:
        return 0
